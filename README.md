from collections import deque

def bfs(grid, start):
    rows, cols = len(grid), len(grid[0])
    queue = deque([start])
    visited = set([start])
    directions = [(0,1), (1,0), (-1,0), (0,-1)]
    
    while queue:
        r, c = queue.popleft()
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols and (nr, nc) not in visited:
                visited.add((nr, nc))
                queue.append((nr, nc))
